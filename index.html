f_ (wip)

Async and modular application development made easy.

Install

npm install f_

Use

const f_ = require('f_');

let f_.getConstructor({
  function_flow: [
    {
      name: 'method1',
      function: function method1() {
        this.f_next();
      }
    },
    {
      name: 'method2',
      function: function method2() {
        this.f_next();
      }
    }
  ]
});
