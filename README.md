## Install

`npm install f_`


## Use

```js
const f_ = require('f_');

let task_list = f_.getConstructor({
  function_flow: [
    {
      name: 'method1',
      function: function method1() {
        this.f_next();
      }
    },
    {
      name: 'method2',
      function: function method2() {
        this.f_next();
      }
    }
  ]
});
```
